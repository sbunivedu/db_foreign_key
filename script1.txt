CREATE TABLE Professors (
    prof_id INT PRIMARY KEY,
    name VARCHAR(50)
);

CREATE TABLE Departments (
    dept_id INT PRIMARY KEY,
    name VARCHAR(50),
    chair_id INT,
    FOREIGN KEY (chair_id)
        REFERENCES Professors(prof_id)
        ON DELETE SET NULL
);
